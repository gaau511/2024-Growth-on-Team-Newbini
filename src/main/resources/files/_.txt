from django.shortcuts import render, redirect
from .forms import ReceiptUploadForm
from .models import ReceiptUpload
from .filehandler import *
import json
import csv

def receiptUpload(request):
    if request.method == 'POST':
        data = request.POST.get('receipt', None)
        receiptupload = ReceiptUpload(
            receipt=data
        )

        

        #file Handler 호출부
        receiptupload.save()
        FileHandler(data)
        return redirect('receiptupload')
    else:
        receiptuploadForm = ReceiptUploadForm
        context = {
            'receiptuploadForm': receiptuploadForm,
        }
        return render(request, 'receiptupload.html', context)
